\section*{Correction -- Exercice 1}

\begin{enumerate}
\item On a $u_0 = 2$ et $u_1 = \dfrac{1}{3}(2) + 4 = \dfrac{14}{3} \approx 4{,}67$.

Donc $u_1 > u_0$. Montrons que $(u_n)$ est croissante.

$u_{n+1} - u_n = \dfrac{1}{3} u_n + 4 - u_n = -\dfrac{2}{3} u_n + 4$

La suite converge vers $\ell = 6$, et tant que $u_n < 6$, on a $u_{n+1} - u_n > 0$.

\textbf{Réponse : (a)} La suite est croissante.

\item $f(x) = x \ln x - x + 1$

$f'(x) = \ln x + x \cdot \dfrac{1}{x} - 1 = \ln x + 1 - 1 = \ln x$

\textbf{Réponse : (a)} $f'(x) = \ln x$

\item On pose $t = e^x$, donc $t > 0$.

$t^2 - 3t + 2 = 0 \iff (t-1)(t-2) = 0$

$t = 1 \Rightarrow e^x = 1 \Rightarrow x = 0$

$t = 2 \Rightarrow e^x = 2 \Rightarrow x = \ln 2$

\textbf{Réponse : (a)} $S = \{0, \ln 2\}$

\item La rotation de centre $O$ et d'angle $\dfrac{\pi}{2}$ multiplie l'affixe par $e^{i\pi/2} = i$.

$z' = i \cdot z_A = i(1 + 2i) = i + 2i^2 = -2 + i$

\textbf{Réponse : (a)} $z' = -2 + i$
\end{enumerate}
